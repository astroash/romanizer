function romanizer(num){
  var result = '';
  var remaining = num;
 
  if (num>= 5000|| num<=0){
    return 'Sorry the Roman\'s didn\'t count that well';
  }
 
  values.forEach(function(value) {
    while (remaining >= value.arabic){
      result += value.roman;
      remaining -= value.arabic;
    }
  });
  return result;
  // example4

};
